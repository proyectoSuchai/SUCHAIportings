570 static int single_frame = 1;
571 int trx_tm_addtoframe(int *data, int len, int mode)
572 {
573  static uint16_t tmframe[TRX_TMFRAMELEN16];
574  static int int16_counter = 0;
575  static int frame_counter = 0;
576  static int tm_type = CMD_STOP;
577  int send_stat = 2;
578 
579  while(mode != CMD_ADDFRAME_EXIT)
580  {
581  switch (mode)
582  {
583  /* A single frame that contains max TRX_TMFRAMELEN bytes of data */
584  case CMD_ADDFRAME_SINGL:
585  /* A new frame being configured */
586  int16_counter = 0;
587  /* Append control field */
588  tmframe[int16_counter++] = CMD_TMFRAME_TSINGL; /* Type (1) */
589  tmframe[int16_counter++] = (uint16_t)(frame_counter); /* Frame# (2) */
590 
591  /* Adding data to the current frame. Max TRX_TMFRAMELEN bytes */
592  while((len > 0) && (int16_counter < TRX_TMFRAMELEN16))
593  {
594  tmframe[int16_counter++] = (uint16_t)(*data); /* Data */
595  data++;
596  len--;
597  }
598 
599  /* Loading frame to TRX and exiting */
600  for(;int16_counter<TRX_TMFRAMELEN16;int16_counter++)
601  {
602  tmframe[int16_counter] = (uint16_t)CMD_STOP;
603  }
604 
605  /* Load and transmit TM */
606  trx_tm_send(tmframe, TRX_TMFRAMELEN16);
607 
608  frame_counter++;
609  int16_counter = 0;
610  mode = CMD_ADDFRAME_EXIT;
611  break;
612 
613  /* A new start frame for more than TRX_TMFRAMELEN bytes, so we need
614  * to load several frames cotaining all the telemetry */
615  case CMD_ADDFRAME_START:
616  /* A new frame being configured */
617  /* Load all remaining frames before */
618  if(int16_counter > 0)
619  {
620  for(;int16_counter<TRX_TMFRAMELEN16;int16_counter++)
621  {
622  tmframe[int16_counter] = (uint16_t)CMD_STOP;
623  }
624 
625  /* Load and transmit TM */
626  trx_tm_send(tmframe, TRX_TMFRAMELEN16);
627  }
628 
629  int16_counter = 0;
630  /* Append control field */
631  tmframe[int16_counter++] = CMD_TMFRAME_TSTART; /* Type (2) */
632  tmframe[int16_counter++] = (uint16_t)(frame_counter); /* Frame# (2) */
633 
634  frame_counter++;
635 
636  /* Adding current tm type */
637  if(len > 0)
638  {
639  tm_type = *data;
640  tmframe[int16_counter++] = (uint16_t)(tm_type); /* DataL */
641 
642  /* Update the remaining data counter */
643  data++;
644  len--;
645  }
646 
647  mode = CMD_ADDFRAME_ADD;
648  break;
649 
650  case CMD_ADDFRAME_CONT:
651  /* A new frame being configured */
652  int16_counter = 0;
653  single_frame = 0;
654  /* Append control field */
655  tmframe[int16_counter++] = CMD_TMFRAME_TCONT; /* Type (2) */
656  tmframe[int16_counter++] = (char)(frame_counter); /* Frame# (2) */
657 
658  /* Add tm type */
659  /* For TM formate please see:
660  * https://docs.google.com/spreadsheets/d/1LKhd_xAB94L9oMWbPKR8nDvijBbtvDN1tCDrdkQpOPM/edit?usp=sharing
661  */
662  //tmframe[int16_counter++] = (uint16_t)(tm_type); /* DataL */ // solo se envia en el START
663 
664  /* Add data if needed */
665  frame_counter++;
666  mode = CMD_ADDFRAME_ADD;
667  break;
668 
669  /* The last frame with the current telemetry */
670  case CMD_ADDFRAME_STOP:
671  /* Load all remaining frames before */
672  if(int16_counter > 0)
673  {
674  for(;int16_counter<TRX_TMFRAMELEN16;int16_counter++)
675  {
676  tmframe[int16_counter] = (uint16_t)CMD_STOP;
677  }
678 
679  /* Load and transmit TM */
680  trx_tm_send(tmframe, TRX_TMFRAMELEN16);
681  }
682 
683  /* A new frame stop being configured */
684  int16_counter = 0;
685  /* Append control field */
686  tmframe[int16_counter++] = CMD_TMFRAME_TSTOP; /* Type (2) */
687  tmframe[int16_counter++] = (uint16_t)(frame_counter); /* Frame# (2) */
688 
689  /* Add data to this frame */
690  frame_counter++;
691  mode = CMD_ADDFRAME_ADD;
692  break;
693 
694  /* Load to TRX all reamaining frames and reset the control field
695  * counters to start a new session. No new data will be append */
696  case CMD_ADDFRAME_FIN:
697  /* Load to trx the remaining frames */
698  if(int16_counter > 0)
699  {
700  for(;int16_counter<TRX_TMFRAMELEN16;int16_counter++)
701  {
702  tmframe[int16_counter] = (uint16_t)CMD_STOP;
703  }
704 
705  /* Load and transmit TM */
706  if(single_frame == 1){
707  tmframe[0] = CMD_TMFRAME_TSINGL;
708  }
709  else{
710  tmframe[0] = CMD_TMFRAME_TSTOP;
711  }
712 
713  trx_tm_send(tmframe, TRX_TMFRAMELEN16);
714  }
715 
716  single_frame = 1;
717  int16_counter = 0;
718  frame_counter = 0;
719  tm_type = CMD_STOP;
720  mode = CMD_ADDFRAME_EXIT;
721  break;
722 
723  /* Append all the requiered data to the current frame or adds more
724  * CONT frame if needed */
725  case CMD_ADDFRAME_ADD:
726  if(len > 0)
727  {
728  /* Current frame is not initilized */
729  if(int16_counter == 0 )
730  {
731  mode = CMD_ADDFRAME_EXIT;
732  }
733  /* Current frame is not full */
734  else if(int16_counter < TRX_TMFRAMELEN16)
735  {
736  /* Adding data to the current frame*/
737  tmframe[int16_counter++] = (uint16_t)(*data); /* DataL */
738  /* Update the remaining data counter */
739  data++;
740  len--;
741  mode = CMD_ADDFRAME_ADD;
742  }
743  else /* Current frame is full */
744  {
745  /* This frame is ready to be loaded into TRX buffer */
746  /* First fill tmframe's unused fields */
747  for(;int16_counter<TRX_TMFRAMELEN16;int16_counter++)
748  {
749  tmframe[int16_counter] = (uint16_t)CMD_STOP;
750  }
751 
752  /* Load and transmit TM */
753  trx_tm_send(tmframe, TRX_TMFRAMELEN16);
754 
755  /* Keep adding data in a new frame */
756  int16_counter = 0;
757  mode = CMD_ADDFRAME_CONT;
758  }
759  }
760  else /* No more data to load */
761  {
762  /* Exit */
763  mode = CMD_ADDFRAME_EXIT;
764  }
765  break;
766 
767  /* Invalid state */
768  default:
769  mode = CMD_ADDFRAME_EXIT;
770  break;
771  }
772  }
773  //return frame_counter;
774 
775  return send_stat;
776 }

785 static int trx_tm_send(uint16_t *data, int len)
786 {
787  #if(SCH_CMDTRX_VERBOSE >= 1)
788  //printf("Sending TM frame...\n");
789  int i; unsigned int h;
790  for(i=0;i<len;i++){
791  h = data[i];
792  printf("0x%04X,", h);
793  }
794  printf("\r\n");
795  #endif
796 
797  int result;
798  result = csp_transaction(CSP_PRIO_NORM, SCH_TRX_NODE_GND, SCH_TRX_PORT_TM,
799  com_timeout, (void *)data, len*2, NULL, 0);
800 
801  return result;
